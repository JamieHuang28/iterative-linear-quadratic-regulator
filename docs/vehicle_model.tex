\section{Non-linear system dynamics linear quadratic planning}
Non-linear system dynamics can be transformed into quadratic form by differentiating around current state. Then the local quadratic problem is solved exactly the same as LQR.
\begin{gather}
    \underbrace{f(x_t, u_t) - f(\hat{x_t}, \hat{u_t})}_{\bar{f}(\delta_{x_t}, \delta_{u_t})} \approx \underbrace{\nabla_{x_t, u_t} f(\hat{u_t}, \hat{u_t})}_{F_t} \begin{bmatrix}
        x_t - \hat{x_t} \\ u_t - \hat{u_t}
    \end{bmatrix} \label{diff-dynamics}
\end{gather}

\section{Kinematic model of vehicle}
Under the assumption that wheels doesn't slip on the ground, the system dynamics is in non-holonomic type. And it is called "vehicle model". The input of the system is $u=[v \quad \delta]^T$.
\begin{equation}
    \begin{bmatrix}
        \dot x\\ \dot y\\ \dot \theta
    \end{bmatrix} = \begin{bmatrix}
        v\cos{\theta}\\
        v\sin{\theta}\\
        \frac{v\tan{\delta}}{L}
    \end{bmatrix} \quad \text{where L is the length of wheel base}
\end{equation}

Descretized form is:
\begin{equation}
    \begin{bmatrix}
        x_{t + 1}\\ y_{t + 1}\\ \theta_{t + 1}
    \end{bmatrix} = \begin{bmatrix}
        x_t + \Delta_t v\cos{\theta}\\
        y_t + \Delta_t v\sin{\theta}\\
        \theta_t + \frac{\Delta_t v\tan{\delta}}{L}
    \end{bmatrix} \label{descret-kinematic-dynamics}
\end{equation}

Referring to \eqref{diff-dynamics}, the kinematic dynamics is transformed:\\
\begin{equation}
\begin{aligned}
    \nabla_{x_t, u_t} f(\hat{x_t}, \hat{u_t}) & = \begin{bmatrix}
        \frac{\partial{f}}{\partial{x_t}} \quad \frac{\partial{f}}{\partial{u_t}}
    \end{bmatrix}\\
    & = \begin{bmatrix}
        1 \quad 0 \quad \Delta_t v \sin{\theta} \quad & \Delta_t \cos{\theta} \quad 0 \\
        0 \quad 1 \quad \Delta_t v \cos{\theta} \quad & \Delta_t \sin{\theta} \quad 0 \\
        0 \quad 0 \quad            1            \quad & \frac{\Delta_t \tan{\delta}}{L} \quad \frac{\Delta_t v}{L\cos{\delta}^2}
    \end{bmatrix}
\end{aligned}
\end{equation}

\section{Augmented model}
To penalize steering rate $\dot{\delta}$ in the cost, we augment the state with $\delta$ and use $\dot{\delta}$ as a control input. The augmented state is $x = [x \quad y \quad \theta \quad \delta]^T$ and input is $u = [v \quad \dot{\delta}]^T$.

Continuous dynamics:
\begin{equation}
    \begin{bmatrix}
        \dot x\\ \dot y\\ \dot \theta\\ \dot \delta
    \end{bmatrix} = \begin{bmatrix}
        v\cos{\theta}\\
        v\sin{\theta}\\
        \frac{v\tan{\delta}}{L}\\
        \dot{\delta}
    \end{bmatrix}
\end{equation}

Discretized form:
\begin{equation}
    \begin{bmatrix}
        x_{t + 1}\\ y_{t + 1}\\ \theta_{t + 1}\\ \delta_{t + 1}
    \end{bmatrix} = \begin{bmatrix}
        x_t + \Delta_t v\cos{\theta}\\
        y_t + \Delta_t v\sin{\theta}\\
        \theta_t + \frac{\Delta_t v\tan{\delta}}{L}\\
        \delta_t + \Delta_t \dot{\delta}
    \end{bmatrix} \label{augmented-discret-dynamics}
\end{equation}

Referring to \eqref{diff-dynamics}, the Jacobians for the augmented model are $A = \frac{\partial f}{\partial x} \in \mathbb{R}^{4 \times 4}$ and $B = \frac{\partial f}{\partial u} \in \mathbb{R}^{4 \times 2}$:
\begin{equation}
\begin{aligned}
    A & = \begin{bmatrix}
        1 & 0 & -\Delta_t v \sin{\theta} & 0 \\
        0 & 1 &  \Delta_t v \cos{\theta} & 0 \\
        0 & 0 & 1 & \frac{\Delta_t v}{L\cos^2{\delta}} \\
        0 & 0 & 0 & 1
    \end{bmatrix}, \quad
    B = \begin{bmatrix}
        \Delta_t \cos{\theta} & 0 \\
        \Delta_t \sin{\theta} & 0 \\
        \frac{\Delta_t \tan{\delta}}{L} & 0 \\
        0 & \Delta_t
    \end{bmatrix}
\end{aligned}
\end{equation}
